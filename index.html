async function exportToOBJ() { // Check if browser supports Streaming File Access (Chrome, Edge, Opera) if (!window.showSaveFilePicker) { const msg = "Streaming Export not supported in this browser. Please use Chrome or Edge on Desktop for large models (5M+ voxels)."; console.error(msg); const box = document.createElement('div'); box.style = "position:fixed;top:50%;left:50%;transform:translate(-50%,-50%);background:#222;color:#ff4444;padding:20px;border:2px solid red;z-index:9999;border-radius:10px;text-align:center;font-family:sans-serif;"; box.innerHTML = `

Export Error

${msg}

`; document.body.appendChild(box); return; } const isMesh = exportMode === 'mesh'; const fileName = isMesh ? 'vox_mesh.obj' : 'vox_boxes.obj'; let fileHandle; try { fileHandle = await window.showSaveFilePicker({ suggestedName: fileName, types: [{ description: 'OBJ Model', accept: { 'text/plain': ['.obj'] } }], }); } catch (e) { console.log("Export cancelled by user"); return; } const writable = await fileHandle.createWritable(); const encoder = new TextEncoder(); const writer = writable.getWriter(); // Write Preamble await writer.write(encoder.encode("# VoxTrotBravo Streaming Export\nmtllib voxel_materials.mtl\no VoxModel\n")); const hs = 0.5; const faceVerts = { px: [new THREE.Vector3( hs,-hs, hs), new THREE.Vector3( hs,-hs,-hs), new THREE.Vector3( hs, hs,-hs), new THREE.Vector3( hs, hs, hs)], nx: [new THREE.Vector3(-hs,-hs,-hs), new THREE.Vector3(-hs,-hs, hs), new THREE.Vector3(-hs, hs, hs), new THREE.Vector3(-hs, hs,-hs)], py: [new THREE.Vector3(-hs, hs, hs), new THREE.Vector3( hs, hs, hs), new THREE.Vector3( hs, hs,-hs), new THREE.Vector3(-hs, hs,-hs)], ny: [new THREE.Vector3(-hs,-hs,-hs), new THREE.Vector3( hs,-hs,-hs), new THREE.Vector3( hs,-hs, hs), new THREE.Vector3(-hs,-hs, hs)], pz: [new THREE.Vector3(-hs,-hs, hs), new THREE.Vector3( hs,-hs, hs), new THREE.Vector3( hs, hs, hs), new THREE.Vector3(-hs, hs, hs)], nz: [new THREE.Vector3( hs,-hs,-hs), new THREE.Vector3(-hs,-hs,-hs), new THREE.Vector3( hs, hs,-hs), new THREE.Vector3( hs, hs,-hs)] }; const dirs = [{d:[1,0,0],v:faceVerts.px}, {d:[-1,0,0],v:faceVerts.nx}, {d:[0,1,0],v:faceVerts.py}, {d:[0,-1,0],v:faceVerts.ny}, {d:[0,0,1],v:faceVerts.pz}, {d:[0,0,-1],v:faceVerts.nz}]; function hasV(gx, gy, gz) { const cx = gx >> 5, cy = gy >> 5, cz = gz >> 5; const k = (cx+65536)+(cy+65536)*131072+(cz+65536)*17179869184; const c = chunks.get(k); return c ? c.voxelMatMap[(gx&31)|((gy&31)<<5)|((gz&31)<<10)] !== -1 : false; } let vertexOffset = 1; const vertMap = isMesh ? new Map() : null; const matNames = ["Voxel_Solid", "Voxel_Glass", "Voxel_Emission", "Voxel_Metal"]; const exportBtn = document.getElementById('export-button-overlay'); const originalText = exportBtn ? exportBtn.innerText : "Export .OBJ"; let voxelCounter = 0; let currentChunkString = ""; for (let m = 0; m < 4; m++) { await writer.write(encoder.encode(`\ng ${matNames[m]}\nusemtl ${matNames[m]}\n`)); for (let [key, chunk] of chunks) { const layer = chunk.layers[m]; if (!layer || layer.count === 0) continue; for (let i = 0; i < layer.count; i++) { const localIdx = layer.indexToLocal[i]; if (localIdx === 65535) continue; const gx = (chunk.cx * 32) + (localIdx & 31); const gy = (chunk.cy * 32) + ((localIdx >> 5) & 31); const gz = (chunk.cz * 32) + (localIdx >> 10); layer.mesh.getColorAt(i, tempColor); if (tempColor.r > 1.0) { tempColor.r /= 255; tempColor.g /= 255; tempColor.b /= 255; } const rS = tempColor.r.toFixed(3), gS = tempColor.g.toFixed(3), bS = tempColor.b.toFixed(3); for (let d = 0; d < 6; d++) { const dir = dirs[d]; if (!hasV(gx + dir.d[0], gy + dir.d[1], gz + dir.d[2])) { const fIdx = []; for (let v = 0; v < 4; v++) { tempPos.copy(dir.v[v]).add({x:gx, y:gy, z:gz}).multiplyScalar(VOXEL_SIZE); const fx = tempPos.x.toFixed(4), fy = tempPos.y.toFixed(4), fz = tempPos.z.toFixed(4); let idx; if (isMesh) { const h = `${fx}_${fy}_${fz}_${rS}_${gS}_${bS}`; idx = vertMap.get(h); if (idx === undefined) { idx = vertexOffset++; vertMap.set(h, idx); currentChunkString += `v ${fx} ${fy} ${fz} ${rS} ${gS} ${bS}\n`; } } else { idx = vertexOffset++; currentChunkString += `v ${fx} ${fy} ${fz} ${rS} ${gS} ${bS}\n`; } fIdx.push(idx); } if (isMesh) { currentChunkString += `f ${fIdx[0]} ${fIdx[1]} ${fIdx[2]}\nf ${fIdx[0]} ${fIdx[2]} ${fIdx[3]}\n`; } else { currentChunkString += `f ${fIdx[0]} ${fIdx[1]} ${fIdx[2]} ${fIdx[3]}\n`; } } } voxelCounter++; // Flush to disk every 1000 voxels to keep RAM usage near zero if (voxelCounter % 1000 === 0) { await writer.write(encoder.encode(currentChunkString)); currentChunkString = ""; if (exportBtn) exportBtn.innerText = `Streaming to Disk: ${Math.round((voxelCounter / totalActiveVoxels) * 100)}%`; await new Promise(resolve => setTimeout(resolve, 0)); } } } } // Final flush and close if (currentChunkString.length > 0) await writer.write(encoder.encode(currentChunkString)); writer.releaseLock(); await writable.close(); if (exportBtn) exportBtn.innerText = originalText; console.log("Export Complete"); }
