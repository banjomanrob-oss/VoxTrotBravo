import bpy

def process_voxel_mesh():
    """
    Cleans up the multi-material imported OBJ generated by 'VoxTrotBravo'.
    Updated for Blender 5.0+ API and EEVEE Next / Cycles node structures.
    Upgraded with full PBR (Physically Based Rendering) node setups including
    Subsurface Scattering, Micro-surface Bump, and Beveled Glass.
    """
    
    selected_objects = [o for o in bpy.context.selected_objects if o.type == 'MESH']
    
    if not selected_objects:
        print("Error: Please select the imported Voxel OBJ mesh(es) first.")
        return

    print(f"Starting cleanup and PBR material mapping on {len(selected_objects)} selected object(s)...")

    # Join objects if multiple are imported (Blender OBJ importer splits by material groups)
    bpy.context.view_layer.objects.active = selected_objects[0]
    if len(selected_objects) > 1:
        bpy.ops.object.join()
        
    obj = bpy.context.active_object

    # --- 1. Geometry Processing (Base Cleanup Only) ---
    bpy.context.view_layer.objects.active = obj
    
    # Determine if this was exported as Mesh (Triangles) or Boxes (Quads) dynamically
    is_mesh_mode = False
    if len(obj.data.polygons) > 0:
        if len(obj.data.polygons[0].vertices) == 3:
            is_mesh_mode = True
            
    if is_mesh_mode:
        print("Mesh mode detected. Merging vertices using optimized C-level operator.")
        bpy.ops.object.mode_set(mode='EDIT')
        bpy.ops.mesh.select_all(action='SELECT')
        bpy.ops.mesh.remove_doubles(threshold=0.001)
        bpy.ops.object.mode_set(mode='OBJECT')
    else:
        print("Boxes mode detected. Leaving geometry as distinct un-merged voxel blocks.")
        
    # OPTIMIZATION: Use C-level 'foreach_set' to apply flat shading to the main grid instantly
    poly_count = len(obj.data.polygons)
    if poly_count > 0:
        obj.data.polygons.foreach_set('use_smooth', [False] * poly_count)

    # --- Enable Shadow Caustics (Cycles 3.2+) ---
    try:
        obj.is_caustics_caster = True
        obj.is_caustics_receiver = True
    except AttributeError:
        pass

    # --- 2. Material Mapping based on OBJ groupings ---
    for mat in obj.data.materials:
        if not mat: continue
        
        print(f"Processing PBR material group: {mat.name}")
        mat.use_nodes = True
        nodes = mat.node_tree.nodes
        links = mat.node_tree.links

        # Clear existing nodes
        nodes.clear()

        # Build Standard Core Nodes
        out_node = nodes.new(type='ShaderNodeOutputMaterial')
        out_node.location = (600, 0)

        bsdf_node = nodes.new(type='ShaderNodeBsdfPrincipled')
        bsdf_node.location = (200, 0)

        attr_node = nodes.new(type='ShaderNodeAttribute')
        attr_node.location = (-150, 200)
        
        # Pull standard vertex color naming
        color_attrs = obj.data.color_attributes
        attr_node.attribute_name = color_attrs[0].name if color_attrs else "Color"

        # Standard linking
        links.new(bsdf_node.outputs['BSDF'], out_node.inputs['Surface'])

        # Contextual Material Routing (Blender 4.0 / 5.0+ Principled BSDF structure)
        if "Solid" in mat.name:
            # 1. Base Color
            links.new(attr_node.outputs['Color'], bsdf_node.inputs['Base Color'])
            
            # 2. Plastic/Resin Sheen (PBR Roughness)
            bsdf_node.inputs['Roughness'].default_value = 0.35 
            
            # 3. Subsurface Scattering (Vital for Photorealism in Voxels)
            if 'Subsurface Weight' in bsdf_node.inputs:
                bsdf_node.inputs['Subsurface Weight'].default_value = 0.15
            elif 'Subsurface' in bsdf_node.inputs:
                bsdf_node.inputs['Subsurface'].default_value = 0.15
                
            # 4. Micro-surface Detail (Plastic Grain)
            noise_node = nodes.new(type='ShaderNodeTexNoise')
            noise_node.location = (-550, -100)
            noise_node.inputs['Scale'].default_value = 250.0 # Fine grain
            noise_node.inputs['Detail'].default_value = 2.0
            
            bump_node = nodes.new(type='ShaderNodeBump')
            bump_node.location = (-350, -100)
            bump_node.inputs['Strength'].default_value = 0.04 # Subtle depth
            bump_node.inputs['Distance'].default_value = 0.1
            links.new(noise_node.outputs['Fac'], bump_node.inputs['Height'])
            
            # 5. Bevel (Soft Edges)
            bevel_node = nodes.new(type='ShaderNodeBevel')
            bevel_node.location = (-150, -100)
            bevel_node.samples = 6
            bevel_node.inputs['Radius'].default_value = 0.008 # Realistic edge rounding
            
            # 6. Stack the Normals: Micro-Bump -> Bevel -> Principled BSDF
            links.new(bump_node.outputs['Normal'], bevel_node.inputs['Normal'])
            links.new(bevel_node.outputs['Normal'], bsdf_node.inputs['Normal'])

        elif "Glass" in mat.name:
            # 1. Base Color & Real Glass Properties
            links.new(attr_node.outputs['Color'], bsdf_node.inputs['Base Color'])
            bsdf_node.inputs['Roughness'].default_value = 0.08 # Clean, slightly frosted finish
            
            if 'IOR' in bsdf_node.inputs:
                bsdf_node.inputs['IOR'].default_value = 1.45 # Standard for glass/acrylic
            
            if 'Transmission Weight' in bsdf_node.inputs:
                bsdf_node.inputs['Transmission Weight'].default_value = 1.0
            elif 'Transmission' in bsdf_node.inputs:
                bsdf_node.inputs['Transmission'].default_value = 1.0
                
            # 2. Beveled Edges for Glass (Crucial for edge-catching light reflections)
            bevel_node = nodes.new(type='ShaderNodeBevel')
            bevel_node.location = (-150, -100)
            bevel_node.samples = 6
            bevel_node.inputs['Radius'].default_value = 0.005
            links.new(bevel_node.outputs['Normal'], bsdf_node.inputs['Normal'])
                
            # Legacy/EEVEE compatibility adjustments
            try:
                mat.blend_method = 'BLEND'
                mat.shadow_method = 'HASHED' 
            except AttributeError:
                pass 

        elif "Emission" in mat.name:
            # Set Base Color to black so the geometry acts strictly as a pure light emitter
            bsdf_node.inputs['Base Color'].default_value = (0, 0, 0, 1) 
            
            # Blender 4.0+ separated Emission Color and Strength
            if 'Emission Color' in bsdf_node.inputs:
                links.new(attr_node.outputs['Color'], bsdf_node.inputs['Emission Color'])
                if 'Emission Strength' in bsdf_node.inputs:
                    bsdf_node.inputs['Emission Strength'].default_value = 10.0
            else:
                links.new(attr_node.outputs['Color'], bsdf_node.inputs['Emission'])

    # --- 3. Viewport Update ---
    for area in bpy.context.screen.areas:
        if area.type == 'VIEW_3D':
            for space in area.spaces:
                if space.type == 'VIEW_3D':
                    space.shading.type = 'MATERIAL'

    # Ensure context updates to reflect shader modifications instantly
    bpy.context.view_layer.update()
    
    print(f"Success! Optimized {obj.name} with PBR-Ready node trees. Render in Cycles for best results!")

if __name__ == "__main__":
    process_voxel_mesh()
